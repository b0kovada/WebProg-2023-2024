document.getElementById('loadCourses').addEventListener('click', () => {
    fetch('https://vvri.pythonanywhere.com/api/courses')
        .then(response => response.json())
        .then(data => {
            const coursesList = document.getElementById('coursesList');
            coursesList.innerHTML = ''; // Listát töröljük, mielőtt új adatokat töltünk be
            data.forEach(course => {
                const listItem = document.createElement('li');
                listItem.textContent = course.name;
                coursesList.appendChild(listItem);
            });
        })
        .catch(error => console.error('Hiba történt: ', error));
});

document.getElementById('newCourseForm').addEventListener('submit', event => {
    event.preventDefault();
    const courseName = document.getElementById('courseName').value;
    fetch('https://vvri.pythonanywhere.com/api/courses', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({ name: courseName })
    })
    .then(response => response.json())
    .then(data => {
        console.log('Kurzus hozzáadva:', data);
        document.getElementById('courseName').value = ''; // Mező törlése
    })
    .catch(error => console.error('Hiba történt: ', error));
});
